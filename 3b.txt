#include<stdio.h>

int max(int a, int b){
    if(a > b)
        return a;
    else
        return b;
}

int main(){

    int n, W;

    printf("Enter number of items: ");
    scanf("%d", &n);

    int weight[100], value[100];

    printf("Enter weights:\n");
    for(int i = 0; i < n; i++){
        scanf("%d", &weight[i]);
    }

    printf("Enter values:\n");
    for(int i = 0; i < n; i++){
        scanf("%d", &value[i]);
    }

    printf("Enter backpack capacity: ");
    scanf("%d", &W);

    int dp[100][100];

    // Initialization
    for(int i = 0; i <= n; i++){

        for(int j = 0; j <= W; j++){

            if(i == 0 || j == 0){
                dp[i][j] = 0;
            }

            else if(weight[i-1] <= j){

                dp[i][j] = max(
                    value[i-1] + dp[i-1][j - weight[i-1]],
                    dp[i-1][j]
                );
            }

            else{
                dp[i][j] = dp[i-1][j];
            }
        }
    }

    printf("\nMaximum Value = %d\n", dp[n][W]);

    printf("\nSelected Items:\n");

    int j = W;
    int totalWeight = 0;

    for(int i = n; i > 0; i--){

        if(dp[i][j] != dp[i-1][j]){

            printf("Item %d -> Weight = %d, Value = %d\n",
                   i, weight[i-1], value[i-1]);

            totalWeight += weight[i-1];

            j = j - weight[i-1];
        }
    }

    printf("\nTotal Weight = %d\n", totalWeight);

    return 0;
}
