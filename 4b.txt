#include<stdio.h>

struct Drama{
    int start, finish, value;
};

int main(){

    int n;

    printf("Enter number of drama schools: ");
    scanf("%d", &n);

    struct Drama d[100];

    printf("Enter start time, finish time and value:\n");

    for(int i = 0; i < n; i++){
        scanf("%d%d%d",
              &d[i].start,
              &d[i].finish,
              &d[i].value);
    }

    int dp[100];

    // Initialize DP array
    for(int i = 0; i < n; i++){
        dp[i] = d[i].value;
    }

    // Dynamic Programming
    for(int i = 1; i < n; i++){

        for(int j = 0; j < i; j++){

            if(d[j].finish <= d[i].start){

                if(dp[j] + d[i].value > dp[i]){

                    dp[i] = dp[j] + d[i].value;
                }
            }
        }
    }

    int maxProfit = 0;

    for(int i = 0; i < n; i++){

        if(dp[i] > maxProfit){
            maxProfit = dp[i];
        }
    }

    printf("\nMaximum Profit = %d\n", maxProfit);

    return 0;
}
