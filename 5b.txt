#include <stdio.h>

#define MAX 10
#define INF 9999

int main()
{
    int graph[MAX][MAX];
    int dist[MAX];
    int V, source;
    int i, j, k;

    printf("Enter number of vertices: ");
    scanf("%d", &V);

    printf("Enter adjacency matrix:\n");

    for(i = 0; i < V; i++)
    {
        for(j = 0; j < V; j++)
        {
            scanf("%d", &graph[i][j]);

            // Replace 0 with INF except diagonal
            if(i != j && graph[i][j] == 0)
            {
                graph[i][j] = INF;
            }
        }
    }

    printf("Enter source vertex: ");
    scanf("%d", &source);

    // Initialize distances
    for(i = 0; i < V; i++)
    {
        dist[i] = INF;
    }

    dist[source] = 0;

    // Bellman-Ford Algorithm
    for(k = 1; k <= V - 1; k++)
    {
        for(i = 0; i < V; i++)
        {
            for(j = 0; j < V; j++)
            {
                if(graph[i][j] != INF &&
                   dist[i] != INF &&
                   dist[i] + graph[i][j] < dist[j])
                {
                    dist[j] = dist[i] + graph[i][j];
                }
            }
        }
    }

    // Check for negative cycle
    for(i = 0; i < V; i++)
    {
        for(j = 0; j < V; j++)
        {
            if(graph[i][j] != INF &&
               dist[i] != INF &&
               dist[i] + graph[i][j] < dist[j])
            {
                printf("\nNegative weight cycle detected\n");
                return 0;
            }
        }
    }

    printf("\nShortest distances from source vertex %d:\n", source);

    for(i = 0; i < V; i++)
    {
        printf("Vertex %d -> Distance = %d\n", i, dist[i]);
    }

    return 0;
}
